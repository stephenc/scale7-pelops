30  	protected Operand(IThriftPool thrift) {
31  		this.thrift = thrift;
32  	}

34  	protected interface IOperation<ReturnType> {
35  		ReturnType execute(IPooledConnection conn) throws Exception;
36  	}
37  
38  	protected <ReturnType> ReturnType tryOperation(IOperation<ReturnType> operation) throws PelopsException {
39  		String lastNode = null;
40  		Exception lastException = null;
41  		int retries = 0;
42  		do {
43  			// Get a connection to a Cassandra node
44              IPooledConnection conn = null;
45              try {
46                  conn = thrift.getConnectionExcept(lastNode);
47              } catch (Exception e) {
48                  // the pool is responsible for blocking and waiting for a connection, so don't retry
49                  throw thrift.getOperandPolicy().getExceptionTranslator().translate(e);
50              }
51  
52              lastNode = conn.getNode().getAddress();
53  			try {
54  				// Execute operation
55                  // Return result!
56  				return operation.execute(conn);
57  			} catch (Exception e) {
58  				// Is this a logic/application error?
59  				if (e instanceof NotFoundException ||
60  					e instanceof InvalidRequestException ||
61  					e instanceof TApplicationException ||
62  					e instanceof AuthenticationException ||
63                      e instanceof TProtocolException ||
64  					e instanceof AuthorizationException) {
65  
66                      // Re-throw application-level exceptions immediately.
67  					throw thrift.getOperandPolicy().getExceptionTranslator().translate(e);
68  				}
69                  // Should we try again?
70                  else if (e instanceof TimedOutException ||
71                      e instanceof TTransportException ||
72                      e instanceof UnavailableException) {
73  
74                      logger.warn("Operation failed as result of network exception. Connection is being marked as corrupt " +
75                              "(and will probably be be destroyed).  See cause for details...", e);
76  
77                      // This connection is "broken" by network timeout or other problem.
78                      conn.corrupted();
79  
80  					retries++;
81  					lastException = e;
82  				}
83                  // nope, throw
84                  else {
85                      throw thrift.getOperandPolicy().getExceptionTranslator().translate(e);
86                  }
87  			} finally {
88                  conn.release();
89              }
90  		} while (retries < thrift.getOperandPolicy().getMaxOpRetries());
91  
92  		throw thrift.getOperandPolicy().getExceptionTranslator().translate(lastException);
93  	}
